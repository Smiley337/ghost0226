## lock.html 비밀번호 설정
👉 https://emn178.github.io/online-tools/sha256.html 에서 비번 해시로 변환 후 아래에 붙여넣기

```js
const HASHED_PASSWORDS = {
  "default": "여기에_해시값_붙여넣기",  // 5개 페이지 공통 비번
  // 페이지별로 다른 비번 원하면:
  // "k9mx2ptf": "AVO_전용_해시값",
  // "r4nw7qvb": "ALIEN_전용_해시값",
};
```
